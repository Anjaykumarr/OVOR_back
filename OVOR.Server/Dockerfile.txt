# =========================================
# 1️⃣ Build stage
# =========================================
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy all project files
COPY ["OVOR.Server/OVOR.Server.csproj", "OVOR.Server/"]
COPY ["../OVOR.Models/OVOR.Models.csproj", "OVOR.Models/"]
COPY ["../OVOR.Repo/OVOR.Repo.csproj", "OVOR.Repo/"]
COPY ["../OVOR.Services/OVOR.Services.csproj", "OVOR.Services/"]

# Restore dependencies
RUN dotnet restore "OVOR.Server/OVOR.Server.csproj"

# Copy all files and build
COPY . .
WORKDIR "/src/OVOR.Server"
RUN dotnet publish -c Release -o /app/publish

# =========================================
# 2️⃣ Runtime stage
# =========================================
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS runtime
WORKDIR /app
COPY --from=build /app/publish .
EXPOSE 8080

# Start the app
ENTRYPOINT ["dotnet", "OVOR.Server.dll"]
